📊 Reporting

Reporting & Analytics APIs

Pull real-time queue stats, historical call data, agent performance metrics, and call detail records into your own dashboards, BI tools, or data warehouses.

What You Can Query

⚡

Real-Time Stats

Live queue depths, agent states, and active contact counts. Refreshed every few seconds.

📅

Historical Reports

Aggregated metrics by day, week, or month. SLA compliance, AHT, abandon rates.

📞

Call Detail Records

One row per contact. Full timeline: queue time, talk time, hold, wrap-up, outcome.

👤

Agent Performance

Per-agent stats: contacts handled, AHT, adherence, state durations.

⚠️

Scope Required

Reporting APIs require the cjp:analyzer_read scope. Historical data is available for up to 13 months. Real-time data is a point-in-time snapshot.

⚡ Real-Time Statistics

Poll real-time endpoints to build live wallboards and supervisor dashboards. Combine with webhooks to push updates on state changes.

⬡ Real-time Queue Snapshot
async function getQueueSnapshot(queueId) {
  const res = await axios.get(
    `https://webexapis.com/v1/contactCenter/queues/${queueId}/statistics`,
    {
      headers: { Authorization: `Bearer ${token}` },
      params:  { orgId: ORG_ID }
    }
  );
  const s = res.data;
  return {
    queueName:        s.queueName,
    contactsWaiting:  s.contactsInQueue,      // waiting for agent
    longestWait:      s.longestContactInQueue, // seconds
    agentsAvailable:  s.agentsAvailable,
    agentsOnContact:  s.agentsOnContact,
    agentsIdle:       s.agentsIdle,
    serviceLevel:     s.serviceLevelPercentage // % in SLA window
  };
}

const snapshot = await getQueueSnapshot('QUEUE_ID');
console.table(snapshot);
⬡ Real-time Agent States
async function getAgentStates(teamId) {
  const res = await axios.get(
    'https://webexapis.com/v1/contactCenter/agents/states',
    {
      headers: { Authorization: `Bearer ${token}` },
      params:  { orgId: ORG_ID, teamId }
    }
  );

  const summary = { Available: 0, Idle: 0, InACall: 0, WrapUp: 0 };
  res.data.items.forEach(a => {
    summary[a.agentState] = (summary[a.agentState] || 0) + 1;
  });
  return summary;
}

// Useful for: team supervisor dashboard, staffing alerts
const states = await getAgentStates('TEAM_ID');
if (states.Available === 0) {
  await sendAlert('⚠️ No agents available in team!');
}
⬡ Auto-polling wallboard (30s interval)
class LiveWallboard {
  constructor(queueIds, intervalMs = 30_000) {
    this.queueIds   = queueIds;
    this.intervalMs = intervalMs;
    this.timer      = null;
  }

  start() {
    this.refresh();
    this.timer = setInterval(() => this.refresh(), this.intervalMs);
    console.log(`📊 Wallboard polling every ${this.intervalMs / 1000}s`);
  }

  stop() {
    clearInterval(this.timer);
    console.log('Wallboard stopped');
  }

  async refresh() {
    const stats = await Promise.all(
      this.queueIds.map(id => getQueueSnapshot(id))
    );
    // Emit to your UI (WebSocket, DOM update, etc.)
    this.emit('update', stats);
  }

  emit(event, data) {
    // Connect to your front-end via WebSocket, EventSource, etc.
    console.log(`[${new Date().toISOString()}]`, data);
  }
}

const board = new LiveWallboard(['QUEUE_1', 'QUEUE_2', 'QUEUE_3']);
board.start();

📅 Historical Reports

Query aggregated metrics for any time range up to 13 months back. Granularity can be hourly, daily, or monthly.

⬡ Historical Queue Metrics
// Get yesterday's stats for all queues
const yesterday = new Date();
yesterday.setDate(yesterday.getDate() - 1);
const dateStr = yesterday.toISOString().split('T')[0]; // "2024-01-14"

const res = await axios.get(
  'https://webexapis.com/v1/contactCenter/reports/queues',
  {
    headers: { Authorization: `Bearer ${token}` },
    params: {
      orgId:       ORG_ID,
      startDate:   `${dateStr}T00:00:00Z`,
      endDate:     `${dateStr}T23:59:59Z`,
      granularity: 'HOURLY',    // HOURLY | DAILY | WEEKLY | MONTHLY
      metrics:     'contactsHandled,avgHandleTime,serviceLevelPercentage,abandonRate'
    }
  }
);

// Each item: { queueId, queueName, hour, contactsHandled, avgHandleTime, ... }
const report = res.data.items;
const totalHandled = report.reduce((sum, r) => sum + r.contactsHandled, 0);
console.log(`Total contacts handled yesterday: ${totalHandled}`);

// Export to CSV
const csv = [
  'Queue,Hour,Handled,AHT (s),SLA %,Abandon %',
  ...report.map(r =>
    `${r.queueName},${r.hour},${r.contactsHandled},${r.avgHandleTime},${r.serviceLevelPercentage},${r.abandonRate}`
  )
].join('\n');
require('fs').writeFileSync('report.csv', csv);

📞 Call Detail Records (CDR)

Every contact generates a CDR — the most granular data available. One record per contact with full timestamps, agent assignments, and outcomes.

⬡ Fetch Call Detail Records
async function fetchCDRs({ startDate, endDate, channelType }) {
  const allRecords = [];
  let cursor = null;

  do {
    const params = { orgId: ORG_ID, startDate, endDate, channelType, max: 1000 };
    if (cursor) params.cursor = cursor;

    const res = await axios.get(
      'https://webexapis.com/v1/contactCenter/recordings/calls',
      { headers: { Authorization: `Bearer ${token}` }, params }
    );

    allRecords.push(...res.data.items);
    cursor = res.data.next ? new URL(res.data.next).searchParams.get('cursor') : null;

  } while (cursor);

  return allRecords;
}

const cdrs = await fetchCDRs({
  startDate:   '2024-01-01T00:00:00Z',
  endDate:     '2024-01-31T23:59:59Z',
  channelType: 'telephony'
});

// Each CDR contains:
// callId, callerId, agentId, agentName, queueId, startTime, endTime,
// talkDuration, holdDuration, wrapUpDuration, totalDuration,
// wrapUpCode, transferCount, disconnectType, callVariables

console.log(`Fetched ${cdrs.length} CDRs`);
const avgHandle = cdrs.reduce((s, c) => s + c.totalDuration, 0) / cdrs.length;
console.log(`Average handle time: ${Math.round(avgHandle)}s`);

📤 Scheduled Data Export

For large-scale data pipelines, schedule nightly exports to your data warehouse instead of polling continuously.

⬡ Nightly ETL Pipeline (Node-cron)
const cron = require('node-cron');
const { BigQuery } = require('@google-cloud/bigquery');

const bq = new BigQuery({ projectId: 'your-gcp-project' });

// Run at 2 AM every night — export yesterday's CDRs to BigQuery
cron.schedule('0 2 * * *', async () => {
  console.log('🔄 Starting nightly WxCC CDR export...');

  const yesterday = new Date(Date.now() - 86_400_000);
  const dateStr   = yesterday.toISOString().split('T')[0];

  const cdrs = await fetchCDRs({
    startDate: `${dateStr}T00:00:00Z`,
    endDate:   `${dateStr}T23:59:59Z`
  });

  if (cdrs.length > 0) {
    await bq
      .dataset('contact_center')
      .table('call_detail_records')
      .insert(cdrs.map(r => ({
        call_id:          r.callId,
        caller_id:        r.callerId,
        agent_id:         r.agentId,
        queue_id:         r.queueId,
        date:             dateStr,
        talk_duration:    r.talkDuration,
        hold_duration:    r.holdDuration,
        total_duration:   r.totalDuration,
        wrap_up_code:     r.wrapUpCode,
        channel_type:     r.channelType
      })));

    console.log(`✅ Exported ${cdrs.length} CDRs to BigQuery`);
  }
});
